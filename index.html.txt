<script>
const $=id=>document.getElementById(id)

async function j(p){const r=await fetch(p);return r.json()}

function showHome(){
$("home").classList.remove("hidden")
$("detail").classList.add("hidden")
}

async function showDetail(slug){
const d=await j(`/data/13f/${slug}.json`)
$("home").classList.add("hidden")
$("detail").classList.remove("hidden")
$("title").textContent=d.name

$("holdings").innerHTML=d.holdings.slice(0,30).map(x=>
`<tr><td>${x.issuer}</td><td>${x.value_usd_k}</td></tr>`).join("")

$("diffs").innerHTML=d.diffs.slice(0,50).map(x=>
`<tr><td>${x.change}</td><td>${x.issuer}</td></tr>`).join("")
}

async function init(){
$("back").onclick=e=>{e.preventDefault();showHome()}
const idx=await j("/data/13f/index.json")
$("subtitle").textContent="updated "+idx.updated_at_utc
$("cards").innerHTML=idx.managers.map(m=>
`<div class="card">
<b>${m.name}</b>
<div>${m.latest_quarter}</div>
<button class="btn" data-slug="${m.slug}">view</button>
</div>`).join("")
document.querySelectorAll("button").forEach(b=>b.onclick=()=>showDetail(b.dataset.slug))
}
init()
</script>
