name: Update 13F data

on:
  workflow_dispatch:

permissions:
  contents: write

jobs:
  run:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-python@v5
        with:
          python-version: "3.11"

      - name: Run
        env:
          SEC_USER_AGENT: ${{ secrets.SEC_USER_AGENT }}
        run: python scripts/fetch_13f.py

      - name: Commit
        run: |
          git config user.name bot
          git config user.email bot@bot
          git add public/data/13f
          git commit -m update || echo no
          git push
